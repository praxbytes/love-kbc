<script>

let selectedAnswer=null;


function flash(color){

    let f=document.getElementById("flash");

    f.className="flash "+color;

    f.style.opacity=1;

    setTimeout(()=>f.style.opacity=0,300);

}


function select(el,val){

    document.querySelectorAll('.option').forEach(o=>o.classList.remove('selected'));

    el.classList.add('selected');

    selectedAnswer=val;

}


function lock(q,correct,reward){

    if(!selectedAnswer) return;


    if(selectedAnswer===correct){

        flash("green");

        document.getElementById("money").innerText=reward;


        setTimeout(()=>{

            document.getElementById("q"+q).classList.remove("active");

            document.getElementById("q"+(q+1)).classList.add("active");

        },400);


    } else {

        flash("red");

        setTimeout(()=>{

            alert("Wrong! Play again 😄");

            location.reload();

        },400);

    }

}


function goFinal(){

    document.getElementById("money").innerText=5;

    document.getElementById("q3").classList.remove("active");

    document.getElementById("q4").classList.add("active");

}


function yes(){

    document.getElementById("money").innerText=7;

    document.getElementById("endScreen").style.display="flex";


    for(let i=0;i<50;i++){

        let c=document.createElement("div");

        c.className="confetti";

        c.style.left=Math.random()*100+"%";

        document.body.appendChild(c);

    }

}
